2. **Working Implementation**:
   ```javascript

private async makeYouTubeRequest(url: string): Promise<Response> {

    const urlWithoutKey = url.replace(/[?&]key=[^&]*/, '');

  } else {
    // Fallback to direct fetch with single key
    return fetch(urlWithKey);
  }
}
```
